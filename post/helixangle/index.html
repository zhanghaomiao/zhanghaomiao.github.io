#ifndef CPPTRAJ_ACTION_HELIXANGLE_H
#define CPPTRAJ_ACTION_HELIXANGLE_H

#include <vector>
#include "Action.h"

class Action_HelixAngle : public Action{
public:
    Action_HelixAngle(): HelixAngle_(nullptr), BB_C_("C"), BB_O_("O"), BB_CA_("CA"),  BB_N_("N"),
    cutOff_{1.5}, debug_(0),  method_{ORIENTATION}{};
    DispatchObject* Alloc() const override {return (DispatchObject* )new Action_HelixAngle();}
    void Help() const override;
private:
    Action::RetType Init(ArgList&, ActionInit&, int) override;
    Action::RetType Setup(ActionSetup&) override;
    Action::RetType DoAction(int, ActionFrame&) override;
    void Print() override {};

    enum MethodType {ORIENTATION=0, ORIENTATION_HBOND, LOOP_ORIENTATION, CAFIT_ORIENTATION};

    class ResidueInfo{
    public:
        ResidueInfo(): C_(-1), CA_(-1), N_(-1), O_(-1), num_(-1){};
        int C()        const {return C_;}
        int CA()       const {return CA_;}
        int N()        const {return N_;}
        int O()        const {return O_;}
        int Num()      const {return num_;}

        void SetC(int i)        {C_  = i;}
        void SetCA(int i)       {CA_ = i;}
        void SetN(int i)        {N_  = i;}
        void SetO(int i)        {O_  = i;}

        bool hasCO() const {return C_ !=-1 && O_ != -1;}
        bool hasNO() const {return N_ !=-1 && O_ != -1;}
        bool hasCN() const {return C_ !=-1 && N_ != -1;}

    private:
        int C_;
        int CA_;
        int N_;
        int O_;
        int num_;
    };

    typedef std::vector<ResidueInfo> ResidueInfoVector;

    ResidueInfoVector MaskToResArray(Topology const&, AtomMask const&) ;
    Vec3 HelixComputeDirection(ActionFrame const&, ResidueInfoVector const&);
    Vec3 calculateDirectionWithCutOff(std::vector<Vec3>& vec3Vector) const;

    AtomMask Mask1_;                         ///<  Mask 1 of helix residue
    AtomMask Mask2_;                         ///<  Mask 2 of helix residue
    DataSet* HelixAngle_;                    ///<  Dataset Hold Helix Angle;
    std::vector<ResidueInfo> Residue1_;      ///<  Residue index which Mask1 contains
    std::vector<ResidueInfo> Residue2_;      ///<  Residue index which Mask2 contains
    NameType BB_C_;                          ///<  Protein N Atom Name ('C')
    NameType BB_O_;                          ///<  Protein N Atom Name ('O')
    NameType BB_CA_;                         ///<  Protein CA Atom Name ('CA')
    NameType BB_N_;                          ///<  Protein N  Atom Name ('N')
    double cutOff_;                          ///<  Drop outliers that is greater than standard deviation * cutOff
    int debug_;                              ///<  Action Debug level
    MethodType method_;                      ///<  Choose which method to calculate Orientation
};

#endif //CPPTRAJ_ACTION_HELIXANGLE_H

void Action_HelixAngle::Help() const {
    mprintf("\t[<name>] <mask1> <mask2> [out <filename>] [method <method>]\n"
            "  Calculate the helix angle between mask1 and mask2 \n"
            "  reference to PyMOl anglebetweenhelices script\n"
            "  mask1 and mask2 must contain more than 4 residues\n"
            "  there are four methods could be used:  \n"
            "  Method: 0, Average direction C(i) -> O(i) pseudo bonds; \n"
            "          1, Average direction O(i) -> N(i+4) hydrogen bonds; \n"
            "          2, Average direction N(i) -> C(i) pseudo bonds; \n"
            "          3, least square linear fit on CA atoms\n");

}

Action::RetType Action_HelixAngle::Init(ArgList& actionArgs, ActionInit &init, int debugIn) {

    debug_ = debugIn;
    DataFile* outfile = init.DFL() .AddDataFile(actionArgs.GetStringKey("out"), actionArgs);
    method_ = static_cast<MethodType>(actionArgs.getKeyInt("method", 0));

    std::string mask1 = actionArgs.GetMaskNext();
    std::string mask2 = actionArgs.GetMaskNext();
    if (mask1.empty() || mask2.empty())  {
        mprinterr("Error: helixAngle: Require 2 masks \n");
        return Action::ERR;
    }
    if(Mask1_.SetMaskString(mask1)) return Action::ERR;
    if(Mask2_.SetMaskString(mask2)) return Action::ERR;

    mprintf("ANGLE: between helix %s - helix %s \n", Mask1_.MaskString(), Mask2_.MaskString());

    HelixAngle_ =  init.DSL().AddSet(DataSet::DOUBLE,
                                     MetaData(actionArgs.GetStringNext(), MetaData::M_ANGLE), "Ang");
    if(!HelixAngle_) return Action::ERR;
    if(outfile) outfile->AddDataSet(HelixAngle_);
    return Action::OK;
}

Action::RetType Action_HelixAngle::Setup(ActionSetup & setup) {
    if(setup.Top().SetupIntegerMask(Mask1_)) return Action::ERR;
    if(setup.Top().SetupIntegerMask(Mask2_)) return Action::ERR;
    Residue1_ = MaskToResArray(setup.Top(), Mask1_);
    Residue2_ = MaskToResArray(setup.Top(), Mask2_);

    if(method_ == ORIENTATION) cutOff_=1.5;
    else if(method_ == ORIENTATION_HBOND) cutOff_=3.5;
    else cutOff_ = 0;

    mprintf("\t");
    Mask1_.BriefMaskInfo();
    Mask2_.BriefMaskInfo();
    mprintf("\t\n");
    if (Residue1_.size() < 4) mprinterr("Error: Mask1: Must contain at least four residues");
    if (Residue2_.size() < 4) mprinterr("Error: Mask2: Must contain at least four residues");
    return Action::OK;
}

Action::RetType Action_HelixAngle::DoAction(int frameNum, ActionFrame & frm) {
    Vec3 dir1 = HelixComputeDirection(frm, Residue1_) ;
    Vec3 dir2 = HelixComputeDirection(frm, Residue2_) ;
    if((dir1[0] == 0 && dir2[1] == 0 && dir1[2] == 0) || (dir2[0] == 0 && dir2[1] == 0 && dir2[2] == 0)) {
        mprintf("\t Can't calculate the orientation for %i as contacts is zero \n", frameNum);
        return Action::SKIP;
    }
    double ang = dir1.Angle(dir2);
    ang *= Constants::RADDEG;
    if (debug_ > 0) {
        dir1.Print("dir1");
        dir2.Print("dir2");
        mprintf("Angle: %g\n", ang);
    }
    HelixAngle_->Add(frameNum, &ang);
    return Action::OK;
}

std::vector<Action_HelixAngle::ResidueInfo>
Action_HelixAngle::MaskToResArray(const Topology & currentParm,
                                  const AtomMask & mask)
{
    int currentResNum = -1;
    std::vector<ResidueInfo> residueInfoVector;
    for(const auto& atom: mask){
        int resNum = currentParm[atom].ResNum();
        if (resNum != currentResNum) {
            Residue const& thisRes = currentParm.Res(resNum);
            ResidueInfo resInfo;
            if(method_ == ORIENTATION)  // only save C and O atom;
            {
                for(int at = thisRes.FirstAtom(); at != thisRes.LastAtom(); at++) {
                    if(currentParm[at].Name() == BB_C_) resInfo.SetC(at);
                    if(currentParm[at].Name() == BB_O_) resInfo.SetO(at);
                }
            }
            else if(method_ == ORIENTATION_HBOND) { // save O and N atom
                for(int at = thisRes.FirstAtom(); at != thisRes.LastAtom(); at++) {
                    if(currentParm[at].Name() == BB_O_) resInfo.SetO(at);
                    if(currentParm[at].Name() == BB_N_) resInfo.SetN(at);
                }
            }

            else if(method_ == LOOP_ORIENTATION) { // Save C and N atom
                for(int at = thisRes.FirstAtom(); at != thisRes.LastAtom(); at++) {
                    if(currentParm[at].Name() == BB_C_) resInfo.SetC(at);
                    if(currentParm[at].Name() == BB_N_) resInfo.SetN(at);
                }
            }

            else if(method_ == CAFIT_ORIENTATION) { // Save CA atom
                for(int at = thisRes.FirstAtom(); at != thisRes.LastAtom(); at++) {
                    if (currentParm[at].Name() == BB_CA_) resInfo.SetCA(at);
                }
            }
            currentResNum = resNum;
            residueInfoVector.push_back(resInfo);
        }
    }
    return residueInfoVector;
}

Vec3 Action_HelixAngle::HelixComputeDirection(const ActionFrame& frm, const ResidueInfoVector&  Residue) {
    int residueNum = Residue.size();
    DataSet_Vector CaData;
    if(method_ == CAFIT_ORIENTATION) {
        for(auto& residueInfo: Residue)  {
            if(residueInfo.CA() != -1) {
                CaData.AddVxyz(frm.Frm().XYZ(residueInfo.CA()));
            }
        }
        Vec3 fit = CaData.LinearFit();
        if(fit * (*(CaData.end()-1) - *CaData.begin()) < 0)
           fit.Neg();
        return fit;
    }

    std::vector<Vec3> vec3Vector;
    if (method_ == ORIENTATION) {
        for (auto &residueInfo: Residue) {
            if (residueInfo.hasCO()) {
                Vec3 v1(frm.Frm().XYZ(residueInfo.C()));
                Vec3 v2(frm.Frm().XYZ(residueInfo.O()));
                vec3Vector.push_back(v1 - v2);
            }
        }
    } else if (method_ == ORIENTATION_HBOND) {
        for (int i = 0; i < residueNum - 4; i++) {
            if (Residue[i].O() != -1 && Residue[i + 4].N() != -1) {
                Vec3 v1(frm.Frm().XYZ(Residue[i].O()));
                Vec3 v2(frm.Frm().XYZ(Residue[i + 4].N()));
                if (DIST2_NoImage(v1, v2) < cutOff_ * cutOff_)
                    vec3Vector.push_back(v1 - v2);
            }
        }
    } else if (method_ == LOOP_ORIENTATION) {
        for (auto &residueInfo: Residue) {
            if (residueInfo.hasCN()) {
                Vec3 v1(frm.Frm().XYZ(residueInfo.C()));
                Vec3 v2(frm.Frm().XYZ(residueInfo.N()));
                vec3Vector.push_back(v1 - v2);
            }
        }
    }
    return calculateDirectionWithCutOff(vec3Vector);
}

Vec3 Action_HelixAngle::calculateDirectionWithCutOff(std::vector<Vec3>& vec3Vector) const {
    Vec3 vec3Sum;
    std::vector<double> angles;
    int residueNum = vec3Vector.size();
    if(residueNum == 0) return {0,0,0};

    for (auto &vec3: vec3Vector) {
        vec3Sum += vec3;
    }

    if (method_ == ORIENTATION) {
        for (auto &vec3: vec3Vector) {
            double angle = vec3Sum.NoNormAngle(vec3);
            angles.push_back(angle);
        }

        auto meanStd = MeanAndStd(angles);
        vec3Sum.Zero();

        for (int i = 0; i < residueNum; ++i) {
            if (std::abs(angles[i] - meanStd.first) < cutOff_ * meanStd.second) {
                vec3Sum += vec3Vector[i];
            } else {
                if(debug_ > 0)
                    mprintf("remove angle %g for it's a outlier. \n", angles[i]);
            }
        }
    }
    vec3Sum.Normalize();
    return vec3Sum;
}
