#ifndef CPPTRAJ_ACTION_FRACTIONNATIVECONTACTS_H
#define CPPTRAJ_ACTION_FRACTIONNATIVECONTACTS_H

#include <map>
#include "Action.h"
#include "ImagedAction.h"

class Action_FractionNativeContacts : public Action {
public:
    Action_FractionNativeContacts(): debug_(0), first_(false){};
    DispatchObject* Alloc() const override{return (DispatchObject*) new Action_FractionNativeContacts();}
    void Help() const override;

private:
    typedef std::vector<int> Iarray;
    typedef std::map<std::pair<int, int>, double> DistInfo;
    Action::RetType Init(ArgList&, ActionInit&, int) override;
    Action::RetType Setup(ActionSetup&) override;
    Action::RetType DoAction(int, ActionFrame&) override;
    int SetupContactLists(Topology const&, Frame const& );
    int DetermineNativeContacts(Topology const&, Frame const&);
    bool ValidContact (int a1, int a2, Topology const& parmIn) const{
        return abs(parmIn[a1].ResNum() - parmIn[a2].ResNum()) >= resoffset_;
    }
    void Print(){}

    DistInfo nativeContact_;           ///< hold nativeContact information
    DataSet* qfrac_;                   ///< hold qfract value
    int resoffset_;                    ///< only calculate the fraction of native contact spaced this far apart
    double distCutoff_;                ///< cutoff in the formula, default value is 4.5 A
    double lambda_;                    ///< lambda in the formula, default value is 1.8
    double beta_;                      ///< beta in the formula, default value is 5
    unsigned int nframes_;             ///< Number of frames
    unsigned int nNativeContacts_;     ///< Number of native contacts
    int debug_;                        ///< action debug level
    bool first_;                       ///< using first frame as reference frame or not
    ImagedAction image_;               ///< Holding imaging-related info/routines
    AtomMask Mask_;                    ///< Only calculate the residue in the mask;
    Iarray contactIdx1_;               ///< Hold atom/residue indicies for Mask
    Topology* CurrentParm_;
    Matrix_3x3 ucell_, recip_;
};


#endif //CPPTRAJ_ACTION_FRACTIONNATIVECONTACTS_H

#include "Action_FractionNativeContacts.h"
#include "CpptrajStdio.h"

void Action_FractionNativeContacts::Help() const {
    mprintf("\t[<mask1>][out <filename>][cutoff <cut>][noimage]\n"
            "\t[ first | %s ]\n"
            "\t[resoffset <n>] [name <dsname>] [beta <beta>] [lambda <lambda>]\n"
            "  Calculate Fraction of native contact (Q) in <mask1> \n"
            "  distance cutoff (4.5 Ang. default), only calculate residue index larger than \n"
            "  offset, default beta is 5 Ang-1, and lambda is 1.8 for all-atom simulation \n"
            "  Equation: Q(X) = 1 / |S| \\sum_{i, j \\in S} \\frac {1}{1+exp(\\beta(r_ij(X) - \\lambda r_{ij}^0)}\n",
            DataSetList::RefArgs);
}

Action::RetType Action_FractionNativeContacts::Init(ArgList& actionArgs, ActionInit& init, int debugIn) {
    debug_ = debugIn;
    double dist_ = actionArgs.getKeyDouble("cutoff", 4.5);
    resoffset_ = actionArgs.getKeyInt("resoffset", 4);  // greater or equal than 4
    beta_ = actionArgs.getKeyDouble("beta", 5);
    lambda_ = actionArgs.getKeyDouble("lambda", 1.8);

    image_.InitImaging(!(actionArgs.hasKey("noimage")));

    if(resoffset_ < 1) {
        mprinterr("Error: Residue offset must be >=0 \n");
        return Action::ERR;
    }

    distCutoff_ = dist_ * dist_ ; // square cutoff
    DataFile* outfile = init.DFL().AddDataFile(actionArgs.GetStringKey("out"), actionArgs);

    ReferenceFrame REF = init.DSL().GetReferenceFrame(actionArgs);
    if(!first_) {
        if (REF.error()) return Action::ERR;
        if (REF.empty()) {
            mprintf("Warinng: No reference structure specified. Defaulting to first.\n");
            first_ = true;
        }
    } else {
        if(!REF.empty())         {
            mprinterr("Error: Must only specify 'first' or a reference structure, not both.\n");
            return Action::ERR;
        }
    }

    std::string name = actionArgs.GetStringKey("name");
    if(name.empty())
        name = init.DSL().GenerateDefaultName("FractionNativeContact");
    qfrac_ = init.DSL().AddSet(DataSet::DOUBLE, MetaData(name, "QFraction"));

    if(outfile != nullptr) {
        outfile -> AddDataSet(qfrac_);
    }

    if(Mask_.SetMaskString(actionArgs.GetMaskNext())) return Action::ERR;
    mprintf("   FractionOfNativeContactS: Mask = '%s'\n", Mask_.MaskString());

    if(!image_.UseImage())
        mprintf("  imaging is off.\n");
    else
        mprintf("  imaging is on. \n");

    if(outfile != nullptr)
        mprintf("\tOutput to '%s'\n", outfile->DataFilename().full());
    mprintf("\tNative Contacts will be ignored for residue spaced < %i apart. \n", resoffset_);
    mprintf("\tNative Contacts will be ignored when distance < %g Angstroms\n", dist_);
    mprintf("\tBeta is set to %g,  and Lambda is set to %g\n", beta_, lambda_);

    if(!first_) {
        image_.SetupImaging(REF.CoordsInfo().TrajBox().Type());
        if(image_.ImageType()==NONORTHO)
            REF.Coord().BoxCrd().ToRecip(ucell_, recip_);
        if(DetermineNativeContacts( REF.Parm(), REF.Coord() )) return Action::ERR;
    }
    return Action::OK;
}

Action::RetType Action_FractionNativeContacts::Setup(ActionSetup & setup) {
    if(SetupContactLists(setup.Top(), Frame()))
        return Action::SKIP;
    mprintf("\t%i potential contact sites for '%s'\n", Mask_.Nselected(), Mask_.MaskString());
    image_.SetupImaging(setup.CoordInfo().TrajBox().Type());
    if(image_.ImagingEnabled())
        mprintf("\tImaging enabled.\n");
    else
        mprintf("\tImaging disabled.\n");
    CurrentParm_ = setup.TopAddress();
    return Action::OK;
}

Action::RetType Action_FractionNativeContacts::DoAction(int frameNum, ActionFrame & frm) {
    if(image_.ImageType() == NONORTHO)     frm.Frm().BoxCrd().ToRecip(ucell_, recip_);
    if(first_) {
        mprintf("\tUsing first frame to determine native contacts.\n");
        if(DetermineNativeContacts(*CurrentParm_, frm.Frm())) return Action::ERR;
        first_ = false;
    }
    nframes_++;
    double sumDist = 0;
    for(int c1 = 0; c1 != Mask_.Nselected(); ++c1){
        for(int c2 = c1 + 1; c2 != Mask_.Nselected(); ++c2)  {
            if(nativeContact_.find({Mask_[c1], Mask_[c2]}) != nativeContact_.end())  {
                double Dist2 = DIST2(frm.Frm().XYZ(Mask_[c1]), frm.Frm().XYZ(Mask_[c2]),
                               image_.ImageType(), frm.Frm().BoxCrd(), ucell_, recip_);
                sumDist += 1.0 / (1 + exp(beta_ * (sqrt(Dist2) - lambda_ *
                        nativeContact_[{Mask_[c1], Mask_[c2]}]) ) );
            }
        }
    }
    sumDist /= nNativeContacts_;
    qfrac_->Add(frameNum, &sumDist);
    return Action::OK;
}

int Action_FractionNativeContacts::SetupContactLists(const Topology &parmIn, const Frame &frameIn) {
    if(parmIn.SetupIntegerMask(Mask_, frameIn)) return 1;
    Mask_.MaskInfo();
    if(Mask_.None()) {
        mprinterr("Warning: Noting selected for '%s' \n", Mask_.MaskString())        ;
        return 1;
    }
    for(int atom : Mask_)
        contactIdx1_.push_back(atom);
    return 0;
}


int Action_FractionNativeContacts::DetermineNativeContacts(const Topology & parmIn, const Frame & fIn) {
    if(SetupContactLists(parmIn, fIn)) return 1;
    for(auto c1 = Mask_.begin(); c1 != Mask_.end(); ++c1)
        for(auto c2 = c1+1; c2!=Mask_.end(); ++c2) {
            if(ValidContact(*c1, *c2, parmIn)) {
                double Dist2 = DIST2(fIn.XYZ(*c1), fIn.XYZ(*c2), image_.ImageType(), fIn.BoxCrd(), ucell_, recip_);
                if(Dist2 < distCutoff_) {
                    nativeContact_.insert({{*c1, *c2}, sqrt(Dist2)});
                }
            }
        }

    nNativeContacts_ = nativeContact_.size();
    mprintf("\tSetup %zu native contacts:\n", nNativeContacts_);
    if(debug_ > 0)
        for (auto & contact : nativeContact_)
        {
            int a1 = contact.first.first;
            int a2 = contact.first.second;
            mprintf("\t\tAtom '%s' to '%s' and dist %g\n", parmIn.AtomMaskName(a1).c_str(),
                    parmIn.AtomMaskName(a2).c_str(), contact.second);
        }
    return 0;
}
