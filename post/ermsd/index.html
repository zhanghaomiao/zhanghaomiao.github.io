#ifndef CPPTRAJ_ACTION_ERMSD_H
#define CPPTRAJ_ACTION_ERMSD_H

#include <vector>
#include "Action.h"
#include "ReferenceAction.h"
#include "DataSet_MatrixDbl.h"

class Action_Ermsd : public Action {
public:
    Action_Ermsd();
    DispatchObject* Alloc() const override {return (DispatchObject*) new Action_Ermsd();}
    void Help() const override;
private:
    Action::RetType Init(ArgList&, ActionInit&, int) override;
    Action::RetType Setup(ActionSetup&) override;
    Action::RetType DoAction(int, ActionFrame&) override;
    void Print() override{};

    Range resRange_;                              ///< Residue range to calculate eRMSD
    DataSet* ermsd_;                              ///< hold ermsd Data
    double cutoff_;                               ///< CUTOFF, default is 2.4 A ,Larger values of cutoff can be useful
                                                   ///< when analyzing unstructured/flexible molecules.
    int debug_;                                   ///< debug level
    int nres_;                                    ///< number of residue (base)
    bool first_;                                  ///< if true, use the first structure as reference
    ReferenceFrame REF_;                          ///< Reference frame to calcuate GRefMat
    typedef std::vector<std::vector<std::vector<double>>> vvv;
    vvv GRefMat_, GActionMat_;                    ///< G matrix for Reference and Action, has shape nres_ * nres_ * 4
    Vec3 F_factors{5, 5, 3};           ///< anisotropic position vector

    class BaseInfo {
    public:
        enum BaseType {None = 0, Purine, Pyrimidine};
        BaseInfo(): C2_(-1), C4_(-1), C6_(-1), idx_(-1), type_(None){}
        BaseInfo(int idx1, int idx2, int idx3, int idx4, BaseType ba):
            C2_(idx1),
            C4_(idx2),
            C6_(idx3),
            idx_(idx4),
            type_(ba){}

        bool checkComplete() const {
            return (C2_ != -1) && (C4_ != -1) && (C6_ != -1) && (idx_ != -1);
        }

        bool operator==(const BaseInfo& rhs) const {
            return type_ == rhs.type_ && checkComplete() && rhs.checkComplete();
        }

        void SetC2(int C2){C2_=C2;}
        void SetC4(int C4){C4_=C4;}
        void SetC6(int C6){C6_=C6;}
        void SetResIdx(int idx){idx_=idx;}
        void SetType(BaseType ba){type_=ba;}
        int C2()           const {return C2_;}
        int C4()           const {return C4_;}
        int C6()           const {return C6_;}
        int ResIdx()       const {return idx_;}
        BaseType type()    const {return type_;}
    private:
        int C2_;
        int C4_;
        int C6_;
        int idx_{};
        BaseType type_;
    };
    std::vector<BaseInfo> vecRefBase_;                  ///< contain baseinfo for reference
    std::vector<BaseInfo> vecCalcBase_;                 ///< contain baseinfo for action frame

    int SetupReference(const Topology&, std::vector<BaseInfo>& );
    void calcGMat(const Frame&, vvv&);
};

#endif

Action_Ermsd::Action_Ermsd():ermsd_(nullptr),cutoff_(0),debug_(0),nres_(0),first_(false){
}

void Action_Ermsd::Help() const {
    mprintf("\t[<name>] [resrange <range>] [out <filename>]\n"
            "\t[cutoff <cutoff>]\n%s",
            ReferenceAction::Help());
    mprintf("   Calculate nucleic acid root-mean-squared deviation of atoms in <mask>\n");
}

Action::RetType Action_Ermsd::Init(ArgList & actionArgs, ActionInit &init, int debugIn) {
    // Initialize keyword
    debug_ = debugIn;
    cutoff_ = actionArgs.getKeyDouble("cutoff", 2.4);
    DataFile* outfile = init.DFL() .AddDataFile(actionArgs.GetStringKey("out"), actionArgs);
    first_ = actionArgs.hasKey("first");

    // Initialize Range
    resRange_.SetRange(actionArgs.GetStringKey("resrange"));
    if(!resRange_.Empty())
        resRange_.ShiftBy(-1);

    // Initialize REF
    REF_ = init.DSL().GetReferenceFrame(actionArgs);
    if(!first_) {
        if (REF_.error()) return Action::ERR;
        if (REF_.empty()) {
            mprintf("Warinng: No reference structure specified. Defaulting to first.\n");
            first_ = true;
        }
    } else {
        if(!REF_.empty())         {
            mprinterr("Error: Must only specify 'first' or a reference structure, not both.\n");
            return Action::ERR;
        }
    }

    ermsd_ = init.DSL().AddSet(DataSet::DOUBLE, MetaData(actionArgs.GetStringNext(),MetaData::M_DISTANCE),
                               "ermsd");
    if(!ermsd_) return Action::ERR;
    if(outfile) outfile -> AddDataSet(ermsd_);

    // Output message
    if(outfile)
        mprintf("\tOutput to '%s'\n",outfile->DataFilename().full());
    mprintf("\tCutoff is set to %g Angstroms\n", cutoff_);
    mprintf("# Citations: Sandro Bottaro, Francesco Di Palma and Giovanni Bussi\n"
            "#            The role of nucleobase interactions in RNA structure and dynamics\n"
            "#            Nucleic Acid Research, 2014, 42, 21, doi: 10.1093/nar/gku972\n");
    return Action::OK;
}

Action::RetType Action_Ermsd::Setup(ActionSetup &setup) {
    // setup Reference Range
    //TODO: if resRange_ is not specified, then choose all the nucleic acid residues
    // as cpptraj dose not have function to decide this, I use the soluteResidues(),
    // this may cause problem. Most of time, please specificy correct resRange_.
    if(resRange_.Empty())
        resRange_ = setup.Top().SoluteResidues();
    if(resRange_.Empty()) {
        mprintf("Waring: No residues specified for %s \n", setup.Top().c_str());
        return Action::SKIP;
    }
    resRange_.PrintRange("\tBase Range", 1);
    mprintf("\n");
    if(SetupReference(setup.Top(), vecCalcBase_)) return Action::ERR;
    if(first_) {
        if (SetupReference(setup.Top(), vecRefBase_))
            return Action::ERR;
    }
    else {
        if (SetupReference(REF_.Parm(), vecRefBase_))
            return Action::ERR;
    }
    if(vecRefBase_.empty() || vecCalcBase_.empty()) {
        mprinterr("Reference or Current Parm contains no base\n");
    }
    if(vecCalcBase_ != vecRefBase_) {
        mprinterr("Reference and Setup Parm does not have same base type");
        return Action::ERR;
    }
    nres_ = vecRefBase_.size();
    mprintf("\n\tERMSD will be calculated on %d bases\n", nres_);
    GRefMat_.resize(nres_, std::vector<std::vector<double>>(nres_, std::vector<double>(4, 0)));
    GActionMat_.resize(nres_, std::vector<std::vector<double>>(nres_, std::vector<double>(4, 0)));
    if(!first_)
        calcGMat(REF_.Coord(), GRefMat_);
    return Action::OK;
}

Action::RetType Action_Ermsd::DoAction(int frameNum, ActionFrame &frm) {
    double res = 0;
    if(first_) {
        mprintf("\tUsing First Frame as a reference structure\n");
        calcGMat(frm.Frm(), GRefMat_);
        first_ = false;
        ermsd_->Add(frameNum, &res);
        return Action::OK;
    }
    calcGMat(frm.Frm(), GActionMat_);
    for(int i = 0; i < nres_; ++i) {
        for(int j = 0; j < nres_; ++j) {
            for(int k = 0; k < 4; ++k) {
                res += (GRefMat_[i][j][k] - GActionMat_[i][j][k])* (GRefMat_[i][j][k] - GActionMat_[i][j][k]);
            }
        }
    }
//     debug GActionMat_;
//    for(int i = 0; i < 10; ++i) {
//        for (int j = 0; j < 10; ++j) {
//            std::cout << GActionMat_[i][j][0] << " ";
//            std::cout << GActionMat_[i][j][1] << " ";
//            std::cout << GActionMat_[i][j][2] << " ";
//            std::cout << GActionMat_[i][j][3] << " ";
//        }
//        std::cout << "\n";
//    }
    res = sqrt(res / nres_);
    ermsd_->Add(frameNum, &res);
    return Action::OK;
}

int Action_Ermsd::SetupReference(const Topology & parmIn, std::vector<BaseInfo>& vecBase) {
    for (auto resnum : resRange_){
        BaseInfo::BaseType basetype_;
        BaseInfo baseinfo;
        //TODO: decide a residue is protein or nucleic acid or ions
        const auto& res = parmIn.Res(resnum);
        const auto c = res.SingleCharName();
        switch(c) {
            case 'A': case 'G':
                basetype_ = BaseInfo::BaseType::Purine;
                break;
            case 'C': case 'T': case 'U':
                basetype_ = BaseInfo::BaseType::Pyrimidine;
                break;
            default:
                basetype_=BaseInfo::BaseType::None;
                mprintf("Warning: not a valid nucleic acid base");
                break;
        }
        if(basetype_) {
            baseinfo.SetType(basetype_);
            for (int i = res.FirstAtom(); i != res.LastAtom(); ++i) {
                auto atomType = parmIn[i].Name();
                if (atomType.Match("C2")) {
                    baseinfo.SetC2(i);
                } else if (atomType.Match("C4")) {
                    baseinfo.SetC4(i);
                } else if (atomType.Match("C6")) {
                    baseinfo.SetC6(i);
                }
            }
            baseinfo.SetResIdx(resnum);
            if(!baseinfo.checkComplete()) {
                mprintf("BaseInfo is not completed, please make sure the nucleic base is correct");
                return 1;
            }
            vecBase.push_back(baseinfo);
        }
    }
    return 0;
}

void Action_Ermsd::calcGMat(const Frame& fIn, vvv& GMat) {
    for(int i=0; i <nres_; ++i)
        for(int j=0; j<nres_; ++j)
            for(int k=0; k<4; ++k)
                GMat[i][j][k] = 0;
    std::vector<Vec3> VecCom;
    std::vector<std::vector<Vec3>> VecDir;
    for(const auto& base: vecRefBase_)  {
        Vec3 C2XYZ{fIn.XYZ(base.C2())};
        Vec3 C4XYZ{fIn.XYZ(base.C4())};
        Vec3 C6XYZ{fIn.XYZ(base.C6())};
        Vec3 COM{(C2XYZ + C4XYZ + C6XYZ) / 3.0};
        Vec3 XDir{C2XYZ-COM};
        XDir.Normalize();
        Vec3 ZDir(0.0);
        if(base.type() == BaseInfo::BaseType::Pyrimidine)
            ZDir = (C4XYZ - COM).Cross(XDir);
        else if(base.type() == BaseInfo::BaseType::Purine)
            ZDir = (C6XYZ - COM).Cross(XDir);
        ZDir.Normalize();
        Vec3 YDir{XDir.Cross(ZDir)};
        YDir.Normalize();
        std::vector<Vec3> Dir = std::vector<Vec3>{XDir, YDir, ZDir};
        VecCom.push_back(COM);
        VecDir.push_back(Dir);
    }

    double firstCutOff = cutoff_ * F_factors[0];
    for(int i = 0; i < nres_; ++i) {
        for(int j = 0; j < nres_; ++j) {
            if (i == j) continue;
            double dist = sqrt((VecCom[i]-VecCom[j]).Magnitude2());
            // check first cutoff
            if(dist > 0.01 && dist < firstCutOff) {
                Vec3 diff = VecCom[j] - VecCom[i];
                Vec3 projection{diff*VecDir[i][0], diff*VecDir[i][1], diff*VecDir[i][2]};
                projection /= F_factors;
                double dist2 = sqrt(projection.Magnitude2());
                double gamma = Constants::PI / cutoff_;
                // second check cutoff
                double sin_gamma_r = sin(gamma * dist2);
                double cos_gamma_r = cos(gamma * dist2);
                if(dist2 <= cutoff_) {
                    GMat[i][j][0] = projection[0] / dist2 * sin_gamma_r / gamma;
                    GMat[i][j][1] = projection[1] / dist2 * sin_gamma_r / gamma;
                    GMat[i][j][2] = projection[2] / dist2 * sin_gamma_r / gamma;
                    GMat[i][j][3] = (1 + cos_gamma_r) / gamma;
                }
            }
        }
    }
}
